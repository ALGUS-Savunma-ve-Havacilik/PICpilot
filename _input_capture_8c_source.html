62 unsigned int* getICValues(unsigned long int sys_time)
63 {
64 
65 #if USE_PPM
66 
71  if ((sys_time - ppm_last_capture_time) > PWM_ALIVE_THRESHOLD){
72  int i = 0;
73  for (i = 0; i < PPM_CHANNELS; i++){
74  capture_value[i] = 0;
75  }
76  }
77  return capture_value;
78 #else
79  int channel;
80  for (channel = 0; channel < 8; channel++) {
81  /*
82  * Check if we received any data from the channel within the threshold.
83  * If not, set the channel input as 0 to indicate that its disconnected
84  */
85  if ((sys_time - last_capture_time[channel]) > PWM_ALIVE_THRESHOLD) {
86  capture_value[channel] = 0;
87  } else { //otherwise the channel is connected. Perform the same necessary calculations
88  if (new_data_available[channel] == 1) {
89  //If the second time is greater than the first time then we have not met roll over
90  if (end_time[channel] > start_time[channel]) {
91  capture_value[channel] = end_time[channel] - start_time[channel];
92  } else {
93  /*
94  * We've reached roll over. Add the maximum time (PR2) to the original start time,
95  * and add on the end time to find the total time. Note that the PR2 register stores
96  * the time period for Timer2, and is set at 20ms. After 20ms, the timer will
97  * reset. Since an average PWM width is 2-3ms max, this is more than sufficient.
98  */
99  capture_value[channel] = ((PR2 - start_time[channel]) + end_time[channel]);
100  }
101  new_data_available[channel] = 0;
102  }
103  }
104  }
105  return capture_value;
106 #endif
107 }
